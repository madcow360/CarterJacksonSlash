\begin{align}
z &= a+ib \hfill a,b \in \mathbb{R}\\
z&=4 \text{ Purely Real}\\
z&=2i \text{ Purely Imaginary}\\
\hline\\
\text{if } z &= -5i+2\\
\Re(z) &= 2\\
\Im(z) &= -5\\
\text{if } w &= 1-i\\
zw &= (-5i+2)(1-i)\\
&= -5i+2-2i+5i^2\\
&= -5i-2i+2-5\\
&= -7i-3\\
\hline
z&= r(\cos(\theta) + i \sin(\theta)) \text{ polar form}\\
r&= |z| = \text{modulus of }z\\
\arg{z} &= \tan^{-1}{\frac{\Re(z)}{\Im(z)}}
\end{align}
